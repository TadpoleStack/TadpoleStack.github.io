        function stopPropagation(evt){
        var e = evt  || window.event;
        if(document.all){//判断是否是IE
            e.cancelBubble = true;
        }else{
            e.stopPropagation();
        }
    }

    function preventDefault(evt){
        var e = evt  || window.event;
        if(e.preventDefault){
            e.preventDefault();
        }else{//针对IE的处理
            e.returnValue=false;
        }
    }
